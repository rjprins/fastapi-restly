Custom Endpoints¶

You can add custom endpoints to your views:

from fastapi_restly import AsyncAlchemyView, include_view

@include_view(app)
class UserView(AsyncAlchemyView):
    prefix = "/users"
    model = User
    schema = UserSchema

    @view_route()
    def custom_endpoint(self):
        return {"message": "This is a custom endpoint"}
